#define _CRT_SECURE_NO_WARNINGS 1

#include <stdio.h>

void print_kjb(int line)
{
	int i = 0;
	for (i = 1; i <=line; i++)
	{
		int j = 0;
		for (j = 1; j <= i; j++)
		{
			printf("%d*%d=%2d ", i, j, i*j);
		}
		printf("\n");
	}
}
int main()
{
	int n = 0;
	scanf("%d", &n);
	print_kjb(n);
	system("pause");
	return 0;
}
